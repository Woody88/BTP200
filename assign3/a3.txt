Prefix::Prefix(const char* filename){
	
	bool rc = false;

		std::ifstream file(filename);
		if (file.is_open()) {
			int no = 0;

				if (file.good()) {
					file >> pubHgh[no]; // version 1.3
					pubLen[no] = std::strlen(pubLow[no]);
					no++;

			}

				this->no = no;
			}
		}
	}

}

	}

}

int Prefix::minNoDigits(int area) const{
	int publisherLen = 0;
	
	for(int i = 0; i < this->no; i++){
		if(this->area[i] == area){
			publisherLen = this->pubLen[i];	
			i = this->no;
		}
	}
	return publisherLen;
}

bool style(std::istream& is, char& s) {
	bool rc = false, ok = false;
	char c;
	do {
		std::cout << " EAN Style ('-', ' ', '\\n' or '0' to quit) : ";
		c = ' ';
		is.get(c);
		if (is.fail()) {
			is.clear();
			is.ignore(2000, '\n');
			std::cerr << " Invalid input. Try again.\n";
        } else if (c != '-' && c != ' ' && c != '\n' && c != '0') {
            is.ignore(2000, '\n');
            std::cerr << " Invalid Character.  Try again.\n";
		} else if (c == '0') {
			if (is.get() != '\n') {
				is.ignore(2000, '\n');
				std::cerr << " Trailing Characters.  Try Again.\n";
			} else
				ok = true;
		} else if (c == '\n') {
			ok = true;
			s = '\0';
			rc = true;
        } else if (is.get() != '\n') {
            is.ignore(2000, '\n');
            std::cerr << " Trailing Characters.  Try Again.\n";
        } else if (c == '-' || c == ' ') {
            ok = true;
			s = c;
			rc = true;
		}
	} while (!ok);

	return rc;
}
